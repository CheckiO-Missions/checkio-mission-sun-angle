Sehen Sie sich diese Lösung an. Mit Hilfe des datetime-Moduls können wir unsere Berechnung in Zeiteinheiten durchführen, und erst dann den Wert in Winkeleinheiten umrechnen. 
import datetime
def sun_angle(time):
    start=datetime.datetime.strptime("06:00","%H:%M")
    end=datetime.datetime.strptime("18:00","%H:%M")
    time=datetime.datetime.strptime(time,"%H:%M")
    diff=(time-start).seconds/60*0.25
    return diff if start<=time<=end else "I don't see the sun!"
